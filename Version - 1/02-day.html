根据一个数字日期，判断这个日期是这一年的第几天
例如： 20160211，计算后结果为42

var num = input01.value;
var year = parseInt(num/10000);  //获取年份
var month = parseInt(num/100)%100; //获取月份
var date = num%100; //获取日期

var sec = 28; //假设2月为28
if( (year%4==0 && year%100!=0) || year%400==0){ //如果是闰年
     sec = 29; //则改为29天
}

var total = 0; //表示一共多少天？
switch(month){
    case 12: total += 30;
    case 11: total += 31;
    case 10: total += 30;
    case 9 : total += 31;
    case 8 : total += 31;
    case 7 : total += 30;
    case 6 : total += 31;
    case 5 : total += 30;
    case 4 : total += 31;
    case 3 : total += sec;
    case 2 : total += 31;

}

console.log("该日期为这一年当中的第"+total+"天");
